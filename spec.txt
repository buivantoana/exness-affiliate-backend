5.2 Luồng Google Ads Thực Tế
Đây là cách toàn bộ hệ thống vận hành trong thực tế khi chạy Google Ads:
SETUP TRONG GOOGLE ADS:
  Final URL:    domain.com/abc          ← trang user thật vào
  Display URL:  domain.com/ExnessReview  ← hiển thị trong ad
LUỒNG GOOGLE BOT (crawl để chấm Quality Score):
  Google crawl domain.com/              ← ReviewPage
  → thấy nội dung editorial sạch, uy tín
  → Quality Score cao → CPC thấp
LUỒNG USER THẬT (click ad):
  User click Google Ad
  → vào domain.com/abc                  ← LandingPage
  → thấy hero video, CTA nổi bật
  → click "Open Account"
  → /go/register?subpath=abc
  → 302 redirect → affiliate link thật
NHIỀU CAMPAIGN TRÊN 1 DOMAIN:
  Campaign Gold   → domain.com/gold     → link gold affiliate
  Campaign Forex  → domain.com/forex    → link forex affiliate
  Campaign VN     → domain.com/vn       → tiếng Việt, link VN

5.4 CatchAllPage — Logic Quan Trọng Nhất
Đây là trái tim của hệ thống. Logic:
Lấy segment đầu tiên của URL: "/abc" → "abc"
Gọi GET /api/sub-paths → lấy danh sách ["abc", "xyz"]
Nếu segment có trong list → render <LandingPage/>
Nếu không → render <NotFoundPage/> (404)
function CatchAllPage() {
  const segment = location.pathname.split("/").filter(Boolean)[0]
  const { data } = useQuery("/api/sub-paths", { staleTime: 60000 })
  if (!data) return <LoadingSpinner/>
  if (data.subPaths.includes(segment)) return <LandingPage/>
  return <NotFoundPage/>
}
Yêu cầu cache: staleTime: 60000 — không gọi API mỗi lần render.
5.5 Hook useLinks()
Hook này cung cấp tất cả affiliate links và hàm navTo() cho mọi component. Tất cả nút trên landing page đều dùng hook này, KHÔNG hardcode link.
function useLinks() {
  const segment = getCurrentSegment() // "abc" hoặc null
  const query = segment ? `?subpath=${segment}` : ""
  const { data } = useQuery(`/api/links${query}`, { staleTime: 60000 })
  function navTo(type) {
    const link = data?.[`${type}Link`] || data?.registerLink || "/"
    window.location.href = link
  }
  return { ...data, navTo }
}

14.3 Logic Backend (Node.js)
Thêm middleware vào server.js, chạy TRƯỚC khi serve static files:
const GEOIP_API = "http://ip-api.com/json"  // 1000 req/phút miễn phí
function getClientIP(req) {
  return req.headers["cf-connecting-ip"]          // Cloudflare
      || req.headers["x-forwarded-for"]?.split(",")[0].trim()
      || req.headers["x-real-ip"]
      || req.socket.remoteAddress
}
async function blockMiddleware(req, res, db, domain) {
  const cfg = getDomainConfig(db, domain)
  const blockedCountries = cfg.blockedCountries || []
  const blockedIPs       = cfg.blockedIPs || []
  // Không block các route API và redirect
  const { pathname } = new URL(req.url, "http://x")
  if (pathname.startsWith("/api") || pathname.startsWith("/go")) return false
  const ip = getClientIP(req)
  // 1. Check IP tĩnh trước (nhanh, không cần API call)
  if (blockedIPs.includes(ip)) return doBlock(res, cfg)
  // 2. Check country qua API
  if (blockedCountries.length > 0 && ip !== "127.0.0.1") {
    try {
      const r = await fetch(`${GEOIP_API}/${ip}?fields=countryCode`)
      const { countryCode } = await r.json()
      if (blockedCountries.includes(countryCode)) return doBlock(res, cfg)
    } catch { /* API lỗi → cho qua */ }
  }
  return false
}
function doBlock(res, cfg) {
  if (cfg.blockAction === "redirect" && cfg.blockRedirectUrl) {
    res.writeHead(302, { Location: cfg.blockRedirectUrl })
    res.end()
  } else if (cfg.blockAction === "drop") {
    res.writeHead(444); res.end()
  } else {
    res.writeHead(403, { "Content-Type": "text/html" })
    res.end("<h1>Access Restricted</h1><p>Not available in your region.</p>")
  }
  return true  // đã xử lý, không chạy tiếp
}
